function generate(length) {
    
    let allowed = "01"
    let result = "";
        
    for(var i=0; i < length; i++){
        
        let rere = rand(0, allowed.length)
        
        result = result + allowed.substring(rere, rere + 1)
        
    }
    
    return result;
    
}

function rand(min, max) {
    
    return Math.floor(Math.random() * (max - min) ) + min;
    
}
